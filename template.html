Selenium for the groovy folks

        def "user is able to login with correct credentials"(){
          given:
          go "/login"

          when:
          $("loginForm").user = "thilko"
          $("loginForm").password = "admin"
          $("#loginButton").click()

          then:
          driver.currentUrl == "/home"
          waitFor{ asyncData.visible }
        }

Page objects

        class LoginPage extends Page {
          static url = "login"
          static content = {
            // some content here
          }
          static at = {
            $("#page_id").value() == "login"
          }
        }
      
        def "user is able to login with correct credentials"(){
          given:
          to LoginPage
          at LoginPage

          when:
          user = "thilko"
          password = "admin"

          then:
          at HomePage
        }
