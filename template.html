Page objects

        class LoginPage extends Page {
          static url = "login"
          static content = {
            // some content here
          }
          static at = {
            $("#page_id").value() == "login"
          }
        }
      
        def "user is able to login with correct credentials"(){
          given:
          to LoginPage
          at LoginPage

          when:
          user = "thilko"
          password = "admin"

          then:
          at HomePage
        }
